>>> Comandos utilizados no projeto <<<
1: npm init
2: yarn add -D typescript
3: yarn add -D @types/node
4: yarn add babel
// Babel: biblioteca responsável pelo build da aplicação e transformação do código tsc em javascript
5: yarn add -D prettier
6: yarn add -D @types/prettier
7: yarn add -D eslint
8: yarn add -D @types/eslint
9: yarn add express
10: yarn add @types/express
11: yarn add typeorm
12: yarn add typeorm-pagination
13: yarn add date-fns
14: yarn add @types/date-fns
15: yarn add express-async-errors
16: yarn add dotenv
17: yarn add @types/dotenv
18: yarn add bcryptjs
19: yarn add @types/bcryptjs
// Bcryptsjs: Biblioteca para criptografia de senhas
20: yarn add cors
21: yarn add @types/cors
22: yarn add rate-limiter-flexible
